#removing unneeded rows
anime_data <- anime_data %>%
  select(anime_id, Score, Genres, Type)
#filtering out for Unknown values and Types not part of this analysis
anime_data <- anime_data %>%
  filter(Score != "UNKNOWN") %>%
  filter(Type != "Music") %>%
  filter(Type != "Movie") %>%
  filter(Type != "OVA") %>%
  filter(Type != "ONA") %>%
  filter(Type != "Special") %>%
  filter(Type != "UNKNOWN") %>%
  filter(Genres != "UNKNOWN")
#converting Score to numeric type
anime_data$Score <- as.numeric(anime_data$Score)
